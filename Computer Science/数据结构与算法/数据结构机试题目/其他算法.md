# 矩阵算法
## 1.马鞍点
[求矩阵中的马鞍点](求矩阵中的马鞍点.pdf)
核心思路：**预处理**
```cpp
#include <iostream>
#include <vector>
#include <climits>  // 为了 INT_MAX
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    // 使用 1-based 索引
    vector<vector<int>> matrix(n + 1, vector<int>(m + 1, 0));
    vector<int> row_min(n + 1, INT_MAX);  // 初始化为很大的数
    vector<int> col_max(m + 1, INT_MIN);  // 初始化为很小的数
    
    // 读入数据并计算每行最小、每列最大
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> matrix[i][j];
            if (matrix[i][j] < row_min[i]) {
                row_min[i] = matrix[i][j];
            }
            if (matrix[i][j] > col_max[j]) {
                col_max[j] = matrix[i][j];
            }
        }
    }
    
    int flag = 0;
    // 遍历每个元素检查是否是马鞍点
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            // 检查是否是行最小且列最大
            if (matrix[i][j] == row_min[i] && matrix[i][j] == col_max[j]) {
                cout << i << " " << j << " " << matrix[i][j] << endl;
                flag = 1;
            }
        }
    }
    
    if (!flag) {
        cout << "NO" << endl;
    }
    
    return 0;
}
```
